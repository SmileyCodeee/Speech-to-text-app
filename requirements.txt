# ---------------------------------------------------------------------------
# Speech-to-Text Note Taking Application — requirements.txt
# Works on both Windows and Linux (macOS too) — no OS-specific packages here.
# Install with:  pip install -r requirements.txt
# ---------------------------------------------------------------------------

# Core UI
streamlit==1.60.0

# Speech recognition (offline) — faster-whisper (CTranslate2 reimplementation
# of OpenAI Whisper): same model weights/accuracy as openai-whisper, but
# 2-4x faster on CPU via int8 quantized inference. Pulls in ctranslate2
# automatically.
faster-whisper==1.2.1

# Audio decoding/normalization — re-encodes and safely decodes audio before
# it reaches the ASR engine, avoiding silent 0-sample decode failures on
# some browser-recorded files.
soundfile==0.14.0
numpy==2.4.6
av==18.0.0

# Portable ffmpeg fallback — used automatically if ffmpeg isn't found on
# the system PATH, so a system-wide ffmpeg install isn't strictly required
# on either Windows or Linux.
imageio-ffmpeg==0.6.0

# Online transcription mode (Google's free Web Speech API) — optional
# alternative to offline faster-whisper, needs internet at transcription time.
SpeechRecognition==3.17.0

# Text processing / keyword extraction (lightweight, multilingual)
yake==0.7.3

# Export formats
python-docx==1.2.0
reportlab==5.0.0

# Optional: abstractive summarization (only needed if you pick "abstractive"
# summarization mode in the sidebar). Loaded directly via
# AutoTokenizer/AutoModelForSeq2SeqLM rather than the pipeline() task
# wrapper, for robustness across transformers versions.
transformers==5.14.1
torch==2.13.0
